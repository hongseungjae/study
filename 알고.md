    static public double comb(double n, double r ){
        double a = 1;
        double b = 1;

        for (double i = n; i > n - r; i--) {
            a = a * i;
        }

        for (double i = r; i > 0; i--) {
            b = b * i;
        }
        
        return a/b;
    }
